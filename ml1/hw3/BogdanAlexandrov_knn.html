def move_up(X):
    tmp = X.reshape(28,28)
    for i in range(tmp.shape[0]):
        if i > 25:
            tmp[i] = np.full(tmp.shape[0], -0.42421296)
        else:
            tmp[i] = tmp[i + 2]
    return tmp.reshape(28*28,)

def move_right(X):
    tmp = X.reshape(28,28)
    for vec in tmp:
        vec[2:] = vec[:-2]
        vec[0] = vec[1] = -0.42421296
    return tmp.reshape(28*28,)

def move_left(X):
    tmp = X.reshape(28,28)
    for vec in tmp:
        vec[:-2] = vec[2:]
        vec[-2] = vec[-1] = -0.42421296
    return tmp.reshape(28*28,)

def move_down(X):
    tmp = X.reshape(28,28)
    for i in range(tmp.shape[0]):
        if i > 25:
            tmp[27 - i] = np.full(tmp.shape[0], -0.42421296)
        else:
            tmp[27 - i] = tmp[27 - i - 2]
    return tmp.reshape(28*28,)

def random_move(X):
    x_ax = np.random.randint(3)
    y_ax = np.random.randint(3)
    if (x_ax == 1):
        move_right(X)
    elif (x_ax == 2):
        move_left(X)
    
    if (y_ax == 1):
        move_up(X)
    elif (y_ax == 2):
        move_down(X)
    return X

diffs = []
m = 10**4

for _ in range(m):
    control_bootstraped = np.random.choice(control, size=n, replace=True)
    test_bootstraped = np.random.choice(test, size=n, replace=True)

    control_mean = np.mean(control_bootstraped)
    test_mean = np.mean(test_bootstraped)

    diffs.append(control_mean - test_mean)
In[]:
alpha = 0.05

diffs_sorted = sorted(diffs)
left_ind = int(m * alpha / 2)
right_ind = int(m * (1 - alpha / 2))
left_point = diffs_sorted[left_ind]
right_point = diffs_sorted[right_ind]
